223unsigned long int random_at_most(long max) {
224 unsigned long int num_bins = (unsigned long) max + 1;
225 unsigned long int num_rand = (unsigned long int) RAND_MAX + 1;
226 unsigned long int bin_size = num_rand / num_bins;
227 unsigned long int defect = num_rand % num_bins;
228
229 unsigned long int x;
230 do
231 x = random();
232 while (num_rand - defect <= (unsigned long)x);
233
234 return x/bin_size;
235}

238u_int8_t getrandtype() {
239 u_int8_t rand = random_at_most(69 - 10 ) + 10; //spare out type 0 (JSON_EMPTY)
240 if(rand <= 9)
241 return 0;
242 if(rand <= 19)
243 return 10;
244 if(rand <= 29)
245 return 20;
246 if(rand <= 39) {
247 rand = random_at_most(2);
248 if(rand == 0)
249 return 30;
250 if(rand == 1)
251 return 31;
252 return 32;
253 }
254 if(rand <= 49)
255 return 40;
256 if(rand <= 59)
257 return 50;
258 if(rand <= 69)
259 return 60;
260 return rand; //low probability for last number!
261}

413int fuzzer(int argc, char *argv[]) {
414 fprintf(stderr, "To test for memory leaks, use e.g.\nvalgrind --leak-check=full -s --exit-on-first-error=yes --error-exitcode=17 %s\n", argv[0]);
415 struct dict* dict = dict_new();
416 struct dict* ret = 0;
417 u_int8_t type = 0;
418 union json_type value, array_value;
419 time_t now, start = time(NULL);
420
421 srandom(start);
422 long long unsigned int loop = 0;
423 long long unsigned int max_loops = 10000000;//4000000000;
424 for(loop = 0; loop<max_loops; loop++) {
425 type = getrandtype();
426 if(loglevel || loop % 100000 == 0) {
427 now = time(NULL);
428 fprintf(stderr, "Loop %lld of %lld with type %s after %ld sec. ", loop, max_loops, json_typetostr(type), now - start);
429 fprintf(stderr, "Percentage of deletes is %.2lf%%. ETA in %lld sec.\n",\
430 (num_del ? ( (double) num_del/(num_upd + num_del) * 100 ) : 0) ,\
431 (long long int) (loop ? ( (double) max_loops/loop * (now-start) - (now - start) ) : -1) );
432 }
433
434 if(loglevel || loop % 1000000 == 0) {
435 now = time(NULL);
436 dict_dump(stdout, dict); fprintf(stdout, "\n");
437 }
438
439 switch(type) { //Fuzz dict_update(...)
440 case JSON_NULL:
441 value.integer = 0;
442 ret = fuzz_dict_update_del(&dict, type, value);
443 break;
444 case JSON_BOOL:
445 value.boolean = true;
446 ret = fuzz_dict_update_del(&dict, type, value);
447 break;
448 case JSON_HEX:
449 //value.hex.number = 0xdeadbeef;
450 value.hex.number = 0x5;
451 value.hex.format = HEX_FORMAT_02; //HEX_FORMAT_04;
452 ret = fuzz_dict_update_del(&dict, type, value);
453 break;
454 case JSON_INT: break;
455 value.integer = 17;
456 ret = fuzz_dict_update_del(&dict, type, value);
457 break;
458 case JSON_FLOAT:

460 ret = fuzz_dict_update_del(&dict, type, value);
461 break;
462 break;
463 case JSON_STR:
464 value.string = malloc(11);
465 snprintf(value.string, 10, "%ld", random_at_most(STR_CONT));
466 ret = fuzz_dict_update_del(&dict, type, value);

468 case JSON_ARRAY: //TODO: Make Arrays also dynamicly tested here
469 value.array = array_new();
470 array_value.integer = 42;

477 array_value.array = array_new();
478 array_add(value.array, JSON_ARRAY, array_value);
479 array_value.object = dict_new();
480 array_add(value.array, JSON_OBJ, array_value);
481 ret = fuzz_dict_update_del(&dict, type, value);
482 break;
483 case JSON_OBJ:
484 value.object = dict_new();

486 break;
487 default: fprintf(stderr,"Unknown Type %u\n", type); break;
488 }
489 if (loglevel) {
490 fprintf(stdout, "Success: %s\n", ret ? "true" : "false");
491 dict_dump(stdout, dict); fprintf(stdout, "\n\n");
492 }
493 if(exit_now) break;
494 }
495 fprintf(stderr, "\n##### FUZZER STOP ##### (in loop %lld with exit_now: %s)\nTime elapsed %ld sec.\n", loop, exit_now ? "true" : "false", now-start);
496 if(!loglevel) {
497 dict_dump(stdout, dict); fprintf(stdout, "\n");
498 }
499 dict_free(dict);
500 return 0;
501}
