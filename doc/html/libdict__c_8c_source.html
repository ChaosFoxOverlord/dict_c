66void dict_printelement(FILE* fp, struct dict* dict) {
67 fprintf(fp, "### PRINT DICT ELEMENT START ###\n");
68 if(dict == 0)
69 fprintf(fp, "\tNull-Pointer provided.");
70 else {
71 fprintf(fp, "\tSelf: %p\n", dict);
72 fprintf(fp, "\tKey: %s\n", dict->key ? dict->key : "(key nil)");
73 fprintf(fp, "\tNext: %p (%s)\n", dict->next, dict->next ? dict->next->key ? dict->next->key : "(key nil)" : "(next nil)");
74 fprintf(fp, "\tprev: %p (%s)\n", dict->prev, dict->prev ? dict->prev->key ? dict->prev->key : "(key nil)" : "(prev nil)");
75 fprintf(fp, "\ttype: %d (%s)\n\tvalue: ", dict->type, json_typetostr(dict->type));
76 switch(dict->type) {
77 case JSON_NULL: fprintf(fp, "null"); break;
78 case JSON_BOOL: fprintf(fp, "%s", dict->value.boolean ? "true" : "false"); break;
79 case JSON_HEX:
80 switch (dict->value.hex.format)
81 {
82 case HEX_FORMAT_STD:
83 case HEX_FORMAT_02:
84 case HEX_FORMAT_04:
85 case HEX_FORMAT_05:
86 case HEX_FORMAT_08:
87 fprintf(fp, __json_hex_format[dict->value.hex.format], dict->value.hex.number); break;
88 break;
89 default:
90 fprintf(fp, __json_hex_format[HEX_FORMAT_STD], dict->value.hex); break;
91 break;
92 }
93 break;
94 case JSON_INT: fprintf(fp, "%lld", dict->value.integer); break;
95 case JSON_FLOAT: fprintf(fp, "%Lf", dict->value.floating); break;
96 case JSON_STR: fprintf(fp, "%s", dict->value.string); break;
97 case JSON_ARRAY: fprintf(fp, "%p", dict->value.array ? dict->value.array : 0); break;
98 case JSON_OBJ: fprintf(fp, "%p (%s)", dict->value.object, dict->value.object ? dict->value.object->key ? dict->value.object->key : "(key nil)" : "(object nil)"); break;
99 default: break;
100 }
101 }
102 fprintf(fp, "\n### PRINT DICT ELEMENT END ###\n");

152struct array* array_add(struct array* array, __uint8_t type, union json_type value) {
153 struct array* new = array;
154 if(array->next == 0) {
155 if(new->type != JSON_EMPTY) new = array_new();
156 array->next = new;
157 new->next = 0;
158 new->type = type;
159 switch(type) {
160 case JSON_NULL: break;
161 case JSON_BOOL: new->value.boolean = value.boolean; break;
162 case JSON_HEX:
163 new->value.hex.number = value.hex.number;
164 new->value.hex.format = value.hex.format;
165 break;
166 case JSON_INT: new->value.integer = value.integer; break;
167 case JSON_FLOAT: new->value.floating = value.floating; break;
168 case JSON_STR:
169 new->value.string = malloc(strlen(value.string)+1);
170 memset(new->value.string, 0, strlen(value.string)+1);
171 strncpy(new->value.string, value.string, strlen(value.string));
172 break;
173 case JSON_ARRAY: new->value.array = value.array; break;
174 case JSON_OBJ: new->value.object = value.object; break;
175 default: new->type = JSON_NULL; break;
176 }
177 new->next = 0;
178 return new;
179 } else {
180 array_add(array->next, type, value);
181 }
182 return NULL;
183}

185struct dict* dict_update(struct dict* dict, __uint8_t type, union json_type value, unsigned int path_len, ...) {
186 if(path_len < 1 || dict == 0) return NULL;
187 va_list valist;
188 char* next_key = 0;
189 struct dict* next_dict = dict;
190 struct dict* old_next_dict = 0;
191 struct dict* last_dict = 0;
192 struct dict* prev_dict = 0;
193
194 //initialize valist
195 va_start(valist, path_len);
196 next_key = va_arg(valist, char *); //get first key of path
197
198 if(path_len == 1) { //Trivial case with path_len == 1
199 //fprintf(stderr, "Trivial\n");
200 last_dict = __dict_plainsearch(dict, next_key, &prev_dict);
201 if(last_dict == 0) { //Element not existing, add new to dict directly
202 return __dict_add(dict, type, value, next_key); //adds new or updates empty
203 } else { //Element existing, empty first, than update
204 return __dict_add(dict_empty(last_dict), type, value, next_key); //adds new or updates empty
205 }
206 }
207
208 //search given path till second last
209 for(int i = 0; i<path_len-1; i++) {
210 //fprintf(stderr, "i: %d\n", i);
211 //fprintf(stderr, "Following Path %s\n", next_key);
212 old_next_dict = next_dict;
213 if(i==0) //first dict has no need to descent
214 next_dict = __dict_plainsearch(next_dict, next_key, &prev_dict);
215 else //decent
216 next_dict = __dict_plainsearch(next_dict->value.object, next_key, &prev_dict);
217 //dict_printelement(stderr, next_dict);
218 if(next_dict == 0) { //Check if path is existing
219 #if DICT_DYN_PATH_UPDATES > 0
220 if(i==0) {
221 //next_dict = __dict_add(dict, JSON_OBJ, (union json_type) dict_new(), next_key); //if not make it existing
222 next_dict = dict_update(dict, JSON_OBJ, (union json_type) dict_new(), 1, next_key);
223 } else {
224 //next_dict = __dict_add(old_next_dict->value.object, JSON_OBJ, (union json_type) dict_new(), next_key); //if not make it existing
225 next_dict = dict_update(old_next_dict->value.object, JSON_OBJ, (union json_type) dict_new(), 1, next_key);
226 }
227
228 if(old_next_dict->value.object == next_dict) //Check if parent dict is linked with last dict, if so it is the first one in a nested dict,...
229 next_dict->prev = old_next_dict; //...thus link it back to parent
230
231 if(type == JSON_OBJ) //Link an inserted nested dict with parent dict
232 next_dict->value.object->prev = next_dict;
233
234 #else
235 va_end(valist); //destroy valist
236 return NULL;
237 #endif
238 }
239 if(next_dict->type != JSON_OBJ){ //Check if a non-JSON Object is on path
240 //fprintf(stderr, "non-JSON Obj on path %s\n", next_key);
241 va_end(valist); //destroy valist
242 return NULL;
243 }
244 next_key = va_arg(valist, char *);
245 }
246
247 va_end(valist); //destroy valist
248
249 //get last key of path by descending in next_dict
250 last_dict = __dict_plainsearch(next_dict->value.object, next_key, &prev_dict);
251
252 if(last_dict != 0){ //if update is exisiting in dict, make it empty first, thus possible included string(s), arrays or dicts must be freed
253 dict_empty(last_dict);
254 }
255
256 //inserting *in* last nested dict in path
257 last_dict = __dict_add(next_dict->value.object, type, value, next_key);
258
259 if(next_dict->value.object == last_dict) //Check if parent dict is linked with last dict, if so it is the first one in a nested dict,...
260 last_dict->prev = next_dict; //...thus link it back to parent
261
262 if(type == JSON_OBJ) //Link an inserted nested dict with parent dict
263 last_dict->value.object->prev = last_dict;
264
265 return last_dict;
266}

268struct dict* __dict_add(struct dict* dict, __uint8_t type, union json_type value, char* key) {
269 struct dict* new = dict;
270 if(dict->next == 0 || dict->type == JSON_EMPTY) { //Last element in list or Empty Element
271 //fprintf(stderr, "INSERT POINT FOUND!\n");
272 if(dict->type != JSON_EMPTY) { //Implicit ... && dict->next == 0, thus end of linked list and new dict-element needed.
273 new = dict_new();
274 new->next = 0;
275 new->prev = dict;
276 dict->next = new;
277 }
278 new->key = malloc(strlen(key)+1); //+1 for \0
279 memset(new->key, 0, strlen(key)+1);
280 strncpy(new->key, key, strlen(key));
281 new->type = type;
282 switch(type) {
283 case JSON_NULL: break;
284 case JSON_BOOL: new->value.boolean = value.boolean; break;
285 case JSON_HEX:
286 new->value.hex.number = value.hex.number;
287 new->value.hex.format = value.hex.format;
288 break;
289 case JSON_INT: new->value.integer = value.integer; break;
290 case JSON_FLOAT: new->value.floating = value.floating; break;
291 case JSON_STR:
292 new->value.string = malloc(strlen(value.string)+1);
293 memset(new->value.string, 0, strlen(value.string)+1);
294 strncpy(new->value.string, value.string, strlen(value.string));
295 break;
296 case JSON_ARRAY: new->value.array = value.array; break;
297 case JSON_OBJ: new->value.object = value.object; break;
298 default: new->type = JSON_NULL; break;
299 }
300 return new;
301 } else {
302 new = __dict_add(dict->next, type, value, key);
303 return new;
304 }
305 return new;
306}

441struct dict* dict_get(struct dict* dict, int path_len, ...) {
442 //fprintf(stderr,"DICT_GET with dict=%p, path_len=%d\n", dict, path_len);
443 va_list valist;
444 char* key = 0;
445 //initialize valist
446 va_start(valist, path_len);
447 if(dict == NULL) {
448 va_end(valist); //destroy valist
449 return NULL;
450 }
451 key = va_arg(valist, char *);
452 while(path_len > 0 && dict != NULL) {
453 //fprintf(stderr, "Searching for key %s...", key);
454 if(dict->key != NULL && strcmp(key, dict->key) == 0) { //found actuall part of path
455 if(path_len == 1) { //Object found
456 //fprintf(stderr, "Object found!\n");
457 return dict;
458 } else if(dict->type == JSON_OBJ) { //Point to descent found
459 dict = dict->value.object;
460 key = va_arg(valist, char *);
461 path_len--;
462 }
463 } else { //Iterate through list, till key ist found
464 dict = dict->next;
465 }
466 }
467 //destroy valist
468 va_end(valist);
469 //if(dict == NULL) fprintf(stderr,"Object *NOT* found!\n");
470 return dict;
471}

532unsigned int __dict_delrec(struct dict* dict, struct dict* first_dict, int path_len, char* next_key, va_list keys) {
533 int retval = DICT_DELREC_NOT_FOUND; // == NULL
534 if(dict == 0 || dict->key == 0) return retval;
535
536 //fprintf(stderr, "__dict_delrec: path_len %d next_key: %s\n", path_len, next_key);
537
538 //fprintf(stderr, "__dict_delrec: path_len %d type: %d key: %s next_key: \n", path_len, dict->type, dict->key);//, next_key);
539
540 bool found = strcmp(next_key, dict->key) == 0 ? true : false;
541
542 if(path_len == 1 && found) {
543 //fprintf(stderr, "DELETE: type: %d key: %s dict->next: %p\n", dict->type, dict->key, dict->next);
544 switch(dict->type) { //update exisiting in dict, handle appropriate
545 case JSON_OBJ:
546 dict_free(dict->value.object);
547 break;
548 case JSON_ARRAY:
549 array_free(dict->value.array);
550 break;
551 case JSON_STR:
552 free(dict->value.string);
553 break;
554 default:
555 break;
556 }
557 //fprintf(stderr, "dict->prev: %s dict->next: %s\n", dict->prev->key, dict->next->key);
558 //fprintf(stderr, "dict->prev %p\n", dict->prev);
559 if(dict->prev) {
560 if(dict->prev->type == JSON_OBJ && dict->prev->value.object == dict) { //check if dict is first element in a nested object
561 //fprintf(stderr, "Is nested!\n");
562 if(!dict->next){ //check if dict is also last element in the nested object
563 dict->prev->value.object = dict_new(); //if so, make new empty dict, so parent element is not left without (empty) content
564 } else
565 dict->prev->value.object = dict->next;
566 } else {
567 //fprintf(stderr, "Is *NOT* nested!\n");
568 dict->prev->next = dict->next;
569 }
570 }
571 if(dict->next) {
572 dict->next->prev = dict->prev;
573 }
574
575 if(dict == first_dict) { //if first element in dict has been deleted...
576 retval = DICT_DELREC_FIRST_DELETED; //...return value is 1
577 } else {
578 retval = DICT_DELREC_OTHER_THEN_FIRST_DELETED; //...else it is 2
579 }
580
581 //fprintf(stderr, "next->prev: %s prev->next: %s\n", dict->next->prev->key, dict->prev->next->key);
582 free(dict->key);
583 free(dict);
584 return retval;
585 } else {
586 if(dict->type == JSON_OBJ && found) {
587 next_key = va_arg(keys, char *);
588 return __dict_delrec(dict->value.object, first_dict, path_len - 1, next_key, keys);
589 } else {
590 return __dict_delrec(dict->next, first_dict, path_len, next_key, keys);
591 }
592 }
593}

595bool dict_del(struct dict** dict_ptr, int path_len, ...) {
596 if(dict_ptr == 0 || *dict_ptr == 0) return false;
597 //fprintf(stderr, "************** dict_del %s\n", (*dict_ptr)->key ? (*dict_ptr)->key : "(nil)");
598 unsigned int dict_delrec = 0;
599 struct dict* next_dict = (*dict_ptr)->next;
600 va_list keys;
601 va_start(keys, path_len);
602 char* next_key = va_arg(keys, char *);
603 dict_delrec = __dict_delrec(*dict_ptr, *dict_ptr, path_len, next_key, keys);

649bool array_del(struct array* array, long int pos) {
650 //fprintf(stderr, "START array_del %p pos: %ld\n", array, pos);
651 if(array == NULL) {
652 //fprintf(stderr,"NOT Deleted!\n");
653 return false;
654 }
655 struct array* array_before = NULL;
656 struct array* array_next = NULL;
657 if(pos == 0) {
658 switch(array->type) {
659 case JSON_EMPTY: return false;
660 case JSON_STR: free(array->value.string); break;
661 case JSON_OBJ: dict_free(array->value.object); break;
662 case JSON_ARRAY: array_free(array->value.array); break;
663 default: break;
664 }
665 if(array->next == NULL) { //last element in array, make (or leave it) JSON_EMPTY
666 memset(array, 0 , sizeof(struct array));
667 //fprintf(stderr,"DELTED LAST! %s (%d)!\n", json_typetostr(array->type), array->type);
668 } else {
669 array_next = array->next; //Copy array to not touch given array pointer, because there may be other references to it.
670 array->type = array_next->type;
671 array->value = array_next->value;
672 array->next = array_next->next;
673 free(array_next); //do not use array free, because values like Strings must be still in place for copy in array!
674 //fprintf(stderr," DELTED Middle!\n");
675
676 }
677 return true;
678 } else {
679 array_before = array_get(array, pos - 1); //iterate array to pos-1
680 if(array_before == NULL)
681 return false;
682 array_next = array_before->next;
683 array_before->next = array_next->next;
684 array_next->next = 0;
685 array_free(array_next);
686 //fprintf(stderr,"DELTED Array!\n");
687 return true;
688 }
689 //fprintf(stderr," Default: NOT Deleted!\n");
690 return false;
691}
